    const handleDeleteMemory = async (memoryId: string) => {
        if (!confirm('Are you sure you want to delete this memory? This action cannot be undone!')) {
            return;
        }

        try {
            const response = await memoryAPI.deleteMemory(memoryId);
            if (response.success) {
                toast.success('Memory deleted successfully');
                setMemories(prev => prev.filter(m => m._id !== memoryId));
            } else {
                toast.error(response.message || 'Failed to delete memory');
            }
        } catch (error: any) {
            toast.error(error.message || 'Failed to delete memory');
        }
    };

    const handleDeleteComment = async (memoryId: string, commentId: string) => {
        if (!confirm('Are you sure you want to delete this comment?')) {
            return;
        }

        try {
            const response = await memoryAPI.deleteComment(memoryId, commentId);
            if (response.success) {
                toast.success('Comment deleted successfully');
                setMemories(prev => prev.map(m =>
                    m._id === memoryId
                        ? { ...m, comments: m.comments.filter(c => c._id !== commentId), commentCount: m.commentCount - 1 }
                        : m
                ));
            } else {
                toast.error(response.message || 'Failed to delete comment');
            }
        } catch (error: any) {
            toast.error(error.message || 'Failed to delete comment');
        }
    };
